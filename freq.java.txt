public class Main
{
	public static void main(String[] args) {
	    int arr[]={1,8,9,7,4};
	    for(int i =0;i<arr.length;i++){
	        int count=1;
	        for(int j=i+1;j<arr.length;j++){
	            if(arr[i]==arr[j]){
	                count=count+1;
	                arr[j]=0;
	            }
	        }
	            if(arr[i]!=0){
	             System.out.println(arr[i]+"freq is "+count); 
	            }
	}
}
}
